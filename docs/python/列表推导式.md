```py
list1 = [2, 4, 6]
list2 = [3, 6, 9]

[3 * x for x in list1] # [6, 12, 18]

[x * y for x in list1 for y in list2] # [6, 12, 18, 12, 24, 36, 18, 36, 54]

[list1[i] * list2[i] for i in range(len(list1))] # [6, 24, 54]
```

```py
# 嵌套列表
matrix = [
	[1, 2, 3],
	[4, 5, 6],
	[7, 8, 9]
]

[[row[i] for row in matrix] for i in range(3)] # [[1, 4, 7], [2, 5, 8], [3, 6, 9]]
```

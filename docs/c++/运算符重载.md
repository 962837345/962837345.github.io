### 加号运算符重载

```cpp
#include <iostream>
#include <string>
using namespace std;

class Person 
{
public:
    int a;

    // 成员函数重载
    Person operator+(Person& p)
    {
        this->a += p.a;
        return *this;
    }
};

// 全局函数重载
//Person operator+(Person& p1, Person& p2)
//{
//    Person temp;
//    temp.a = p1.a + p2.a;
//    return temp;
//}

void test()
{
    Person p1;
    p1.a = 10;

    Person p2;
    p2.a = 20;

    Person p3 = p1 + p2;
    cout << "P3 = " << p3.a << endl;
}


int main()
{
    test();

    system("pause");

    return 0;
}
```

### 左移运算符重载

```cpp
#include <iostream>
#include <string>
using namespace std;

class Person 
{
    friend ostream& operator<<(ostream& cout, Person p);
public:
    Person(int a, int b)
    {
        this->a = a;
        this->b = b;
    }

private:
    int a;
    int b;
    
    // 左移运算符无法通过成员函数重载
};

// 全局函数重载
ostream& operator<<(ostream &cout, Person p)
{
    cout << "a = " << p.a << " b = " << p.b << endl;
    return cout;
}

void test()
{
    Person p1(10, 20);
    cout << p1;
}


int main()
{
    test();

    system("pause");

    return 0;
}
```

### 递增运算符重载

```cpp
#include <iostream>
#include <string>
using namespace std;

class Person
{
    friend ostream& operator<<(ostream& cout, Person p);
public:
    int a = 0;

    Person& operator++()
    {
        a++;
        return *this;
    }

    // int代表占位，区分前置和后置
    // 返回值不能是一个引用
    Person operator++(int)
    {
        Person temp = *this;
        a++;
        return temp;
    }
};

ostream& operator<<(ostream& cout, Person p)
{
    cout << "a = " << p.a << endl;
    return cout;
}

void test()
{
    Person p;
    cout << p++ << endl;
    cout << p << endl;
}


int main()
{
    test();

    system("pause");

    return 0;
}
```

### 赋值运算符重载

```cpp
#include <iostream>
#include <string>
using namespace std;

class Person
{
public:
    int * a;
    Person(int a)
    {
        this->a = new int(a);
    }

    ~Person()
    {
        if (a != NULL)
        {
            delete a;
            a = NULL;
        }
    }

    Person& operator=(Person& p)
    {
        // 这里需要先判断是否有属性在堆区，如果有先释放干净，然后再深拷贝 
        if (a != NULL)
        {
            delete a;
            a = NULL;
        }

        a = new int(*p.a);

        return *this;
    }

};


void test()
{
    Person p1(10);
    Person p2(20);
    Person p3(30);
    
    p3 = p2 = p1;

    cout << "p1 a = " << *p1.a << endl;
    cout << "p2 a = " << *p2.a << endl;
    cout << "p3 a = " << *p3.a << endl;
}


int main()
{
    test();

    system("pause");

    return 0;
}
```

### 关系运算符重载

```cpp
#include <iostream>
#include <string>
using namespace std;

class Person
{
public:
   
    Person(string name, int age)
    {
        this->name = name;
        this->age = age;
    }

    bool operator==(Person& p)
    {
        return this->name == p.name && this->age == p.age;
    }

    bool operator!=(Person& p)
    {
        return !(this->name == p.name && this->age == p.age);
    }

    string name;
    int age;
};


void test()
{
    Person p1("小明", 18);
    Person p2("小明", 18);

    if (p1 == p2)
    {
        cout << "p1与p2相等 "<< endl;
    }
    else
    {
        cout << "p1与p2不相等 " << endl;
    }

}


int main()
{
    test();

    system("pause");

    return 0;
}
```

### 函数调用运算符重载

```cpp
#include <iostream>
#include <string>
using namespace std;

class Person
{
public:
   
    void operator()(string text)
    {
        cout << text << endl;
    }
};


void test()
{
    Person p;
    p("Print text");

    // 匿名函数调用
    Person()("Print text");
}


int main()
{
    test();

    system("pause");

    return 0;
}
```

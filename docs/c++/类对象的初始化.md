普通的初始化操作

```cpp
#include <iostream>
#include <string>
using namespace std;

class Person 
{
public:
    int a;
    int b;
    int c;
    
    Person(int a, int b, int c)
    {
        this->a = a;
        this->b = b;
        this->c = c;
    }
    // 带初始值的构造函数
    // Person(int a = 1, int b = 2, int c =3)
    // {
    //     this->a = a;
    //     this->b = b;
    //     this->c = c;
    // }
};

void test()
{
    Person p1(10, 20, 30);
    cout << "a = " << p1.a << endl;
    cout << "b = " << p1.b << endl;
    cout << "c = " << p1.c << endl;
}


int main()
{
    test();

    system("pause");

    return 0;
}

```

C++中支持的另一种初始化形式

```cpp
#include <iostream>
#include <string>
using namespace std;

class Person 
{
public:
    int a;
    int b;
    int c;
    
    Person(int a, int b, int c) : a(a), b(b), c(c)
    {

    }
};

void test()
{
    Person p1(10, 20, 30);
    cout << "a = " << p1.a << endl;
    cout << "b = " << p1.b << endl;
    cout << "c = " << p1.c << endl;
}


int main()
{
    test();

    system("pause");

    return 0;
}
```

## 最普通使用两次for循环
```js
let arr = [1,2,3,3,5,2,4,1,2,3,2,4,6];
for(let i = 0; i < arr.length; i++){
  for(let j = i+1; j< arr.length; j++){
    if(arr[i] === arr[j]){
      arr.splice(j,1);
      j--;
    }
  }
}
console.log(arr)
```
输出 `[1, 2, 3, 5, 4, 6]`

## 使用ES6的from方法
```js
let arr = [1,2,3,3,5,2,4,1,2,3,2,4,6];
Array.from(new Set(arr));
console.log(arr)
```
输出 `[1, 2, 3, 5, 4, 6]` 

## 使用indexOf去重
```js
let arr = [1,2,3,3,5,2,4,1,2,3,2,4,6];
let array = [];
for(let i = 0; i < arr.length; i++){
  if(array.indexOf(arr[i]) === -1){
    array.push(arr[i])
  }
}
console.log(array)
```
输出 `[1, 2, 3, 5, 4, 6]` 

## 使用sort去重
```js
let arr = [1,2,3,3,5,2,4,1,2,3,2,4,6];
arr = arr.sort();
let array = [arr[0]];
for(let i = 1; i < arr.length; i++){
  if(arr[i] !== arr[i-1]){
    array.push(arr[i])
  }
}
console.log(array)
```
输出 `[1, 2, 3, 4, 5, 6]` 

## 使用reduce去重
```js
let arr = [1,2,3,3,5,2,4,1,2,3,2,4,6];
let array = arr.reduce((prev,cur) => {
  if(!prev.includes(cur)){
    prev.push(cur)
  }
  return prev;
},[]);

console.log(array)
```
